Employee Salary Slip Program


#include 
//Program to Store Employee Details & Generate Salary Slip//
struct employee {
    char name[30];
    int age;
    char designation[30];
    float basic;
    float ta, da, hra, pf, gross, net;
};

int main() {
    struct employee e[50];
    int n, i;

    printf("Enter number of employees: ");
    scanf("%d", &n);

    for(i = 0; i < n; i++) {
        printf("\nEnter details for Employee %d\n", i + 1);

        printf("Name: ");
        scanf("%s", e[i].name);

        printf("Age: ");
        scanf("%d", &e[i].age);

        printf("Designation: ");
        scanf("%s", e[i].designation);

        printf("Basic Salary: ");
        scanf("%f", &e[i].basic);

        // Salary calculations
        e[i].ta  = 0.35 * e[i].basic;
        e[i].da  = 0.60 * e[i].basic;
        e[i].hra = 0.33 * e[i].basic;
        e[i].pf  = 0.12 * e[i].basic;

        e[i].gross = e[i].basic + e[i].ta + e[i].da + e[i].hra;
        e[i].net   = e[i].gross - e[i].pf;
    }

    // Display salary slip
    printf("\n\n========= SALARY SLIP =========\n");

    for(i = 0; i < n; i++) {
        printf("\nEmployee %d:\n", i + 1);
        printf("Name        : %s\n", e[i].name);
        printf("Age         : %d\n", e[i].age);
        printf("Designation : %s\n", e[i].designation);
        printf("Basic       : %.2f\n", e[i].basic);

        printf("TA (35%)    : %.2f\n", e[i].ta);
        printf("DA (60%)    : %.2f\n", e[i].da);
        printf("HRA (33%)   : %.2f\n", e[i].hra);
        printf("PF (12%)    : %.2f\n", e[i].pf);

        printf("Gross Salary: %.2f\n", e[i].gross);
        printf("Net Salary  : %.2f\n", e[i].net);

        printf("-----------------------------------\n");
    }

    return 0;
}
